Template 1: File Processor MCP

A complete MCP server that reads, searches, and analyzes files. Perfect for building code review bots, documentation generators, or migration helpers.

#!/usr/bin/env node import { Server } from "@modelcontextprotocol/sdk/server/index.js"; import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js"; import { CallToolRequestSchema, ListToolsRequestSchema, } from "@modelcontextprotocol/sdk/types.js"; import { readFileSync, writeFileSync, readdirSync, statSync } from "fs"; import { join, resolve } from "path"; const server = new Server( { name: "file-processor", version: "1.0.0" }, { capabilities: { tools: {} } } ); // Define tools server.setRequestHandler(ListToolsRequestSchema, async () => ({ tools: [ { name: "read_file", description: "Read contents of a file", inputSchema: { type: "object", properties: { path: { type: "string", description: "File path to read" } }, required: ["path"] } }, { name: "search_files", description: "Search for files matching a pattern", inputSchema: { type: "object", properties: { directory: { type: "string", description: "Directory to search" }, pattern: { type: "string", description: "Regex pattern to match" } }, required: ["directory", "pattern"] } }, { name: "analyze_structure", description: "Analyze directory structure and file types", inputSchema: { type: "object", properties: { path: { type: "string", description: "Root directory" } }, required: ["path"] } } ] })); // Handle tool calls server.setRequestHandler(CallToolRequestSchema, async (request) => { const { name, arguments: args } = request.params; if (name === "read_file") { const content = readFileSync(resolve(args.path), "utf-8"); return { content: [{ type: "text", text: content }] }; } if (name === "search_files") { const regex = new RegExp(args.pattern); const results = []; function walk(dir) { for (const entry of readdirSync(dir)) { const full = join(dir, entry); const stat = statSync(full); if (stat.isDirectory()) walk(full); else if (regex.test(entry)) results.push(full); } } walk(resolve(args.directory)); return { content: [{ type: "text", text: results.join("\n") || "No matches" }] }; } if (name === "analyze_structure") { const stats = {}; function walk(dir) { for (const entry of readdirSync(dir)) { const full = join(dir, entry); const stat = statSync(full); if (stat.isDirectory()) walk(full); else { const ext = entry.split(".").pop() || "none"; stats[ext] = (stats[ext] || 0) + 1; } } } walk(resolve(args.path)); return { content: [{ type: "text", text: JSON.stringify(stats, null, 2) }] }; } throw new Error("Unknown tool: " + name); }); // Start server const transport = new StdioServerTransport(); await server.connect(transport);

Template 2: API Client MCP

Wrap any REST API as an MCP server. Your AI gets direct access to external services.

#!/usr/bin/env node import { Server } from "@modelcontextprotocol/sdk/server/index.js"; import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js"; import { CallToolRequestSchema, ListToolsRequestSchema, } from "@modelcontextprotocol/sdk/types.js"; const BASE_URL = process.env.API_BASE_URL || "https://api.example.com"; const API_KEY = process.env.API_KEY || ""; const server = new Server( { name: "api-client", version: "1.0.0" }, { capabilities: { tools: {} } } ); async function apiRequest(method, endpoint, body) { const res = await fetch(BASE_URL + endpoint, { method, headers: { "Content-Type": "application/json", ...(API_KEY ? { Authorization: "Bearer " + API_KEY } : {}) }, ...(body ? { body: JSON.stringify(body) } : {}) }); if (!res.ok) throw new Error(res.status + ": " + await res.text()); return res.json(); } server.setRequestHandler(ListToolsRequestSchema, async () => ({ tools: [ { name: "api_get", description: "GET request to the API", inputSchema: { type: "object", properties: { endpoint: { type: "string", description: "API endpoint path" } }, required: ["endpoint"] } }, { name: "api_post", description: "POST request to the API", inputSchema: { type: "object", properties: { endpoint: { type: "string", description: "API endpoint path" }, data: { type: "object", description: "Request body" } }, required: ["endpoint", "data"] } } ] })); server.setRequestHandler(CallToolRequestSchema, async (request) => { const { name, arguments: args } = request.params; if (name === "api_get") { const data = await apiRequest("GET", args.endpoint); return { content: [{ type: "text", text: JSON.stringify(data, null, 2) }] }; } if (name === "api_post") { const data = await apiRequest("POST", args.endpoint, args.data); return { content: [{ type: "text", text: JSON.stringify(data, null, 2) }] }; } throw new Error("Unknown tool: " + name); }); const transport = new StdioServerTransport(); await server.connect(transport);

Template 3: Data Store MCP

Persistent JSON-based key-value store with search. Perfect for building knowledge bases, caches, or config managers.

#!/usr/bin/env node import { Server } from "@modelcontextprotocol/sdk/server/index.js"; import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js"; import { CallToolRequestSchema, ListToolsRequestSchema, } from "@modelcontextprotocol/sdk/types.js"; import { readFileSync, writeFileSync, existsSync, mkdirSync } from "fs"; import { join } from "path"; const STORE_DIR = process.env.STORE_PATH || join(process.env.HOME, ".mcp-store"); if (!existsSync(STORE_DIR)) mkdirSync(STORE_DIR, { recursive: true }); const DB_FILE = join(STORE_DIR, "data.json"); function loadDB() { if (!existsSync(DB_FILE)) return {}; return JSON.parse(readFileSync(DB_FILE, "utf-8")); } function saveDB(db) { writeFileSync(DB_FILE, JSON.stringify(db, null, 2)); } const server = new Server( { name: "data-store", version: "1.0.0" }, { capabilities: { tools: {} } } ); server.setRequestHandler(ListToolsRequestSchema, async () => ({ tools: [ { name: "store_set", description: "Store a key-value pair", inputSchema: { type: "object", properties: { key: { type: "string" }, value: { type: "string" }, tags: { type: "array", items: { type: "string" }, description: "Optional tags" } }, required: ["key", "value"] } }, { name: "store_get", description: "Retrieve a value by key", inputSchema: { type: "object", properties: { key: { type: "string" } }, required: ["key"] } }, { name: "store_search", description: "Search entries by keyword or tag", inputSchema: { type: "object", properties: { query: { type: "string" } }, required: ["query"] } }, { name: "store_list", description: "List all keys", inputSchema: { type: "object", properties: {} } } ] })); server.setRequestHandler(CallToolRequestSchema, async (request) => { const { name, arguments: args } = request.params; const db = loadDB(); if (name === "store_set") { db[args.key] = { value: args.value, tags: args.tags || [], updated: new Date().toISOString() }; saveDB(db); return { content: [{ type: "text", text: "Stored: " + args.key }] }; } if (name === "store_get") { const entry = db[args.key]; if (!entry) return { content: [{ type: "text", text: "Not found: " + args.key }] }; return { content: [{ type: "text", text: JSON.stringify(entry, null, 2) }] }; } if (name === "store_search") { const q = args.query.toLowerCase(); const results = Object.entries(db).filter(([k, v]) => k.toLowerCase().includes(q) || v.value.toLowerCase().includes(q) || (v.tags || []).some(t => t.toLowerCase().includes(q)) ); return { content: [{ type: "text", text: results.length ? results.map(([k, v]) => k + ": " + v.value.substring(0, 100)).join("\n") : "No results for: " + args.query }] }; } if (name === "store_list") { const keys = Object.keys(db); return { content: [{ type: "text", text: keys.length ? keys.join("\n") : "Store is empty" }] }; } throw new Error("Unknown tool: " + name); }); const transport = new StdioServerTransport(); await server.connect(transport);

Building custom tools

Every MCP tool has 3 parts: a name, a description (the AI reads this to decide when to use it), and an inputSchema (JSON Schema for parameters).

Best practices:
• Write clear descriptions — the AI uses them to choose tools
• Use Zod for input validation in production
• Return structured JSON for complex data
• Handle errors gracefully with try/catch
• Add "type": "module" to package.json for ES imports
